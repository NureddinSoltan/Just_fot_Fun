{
    stack<char>s;
    string posfix="";
    for (int i=0;i<t.size();i++)
    {
        if (t[i]>=0&&t[i]<=9)
        {
             posfix+=t[i];
        }
        else
        {
            while (!s.empty()&& prec(s.top())>=prec(t[i]))
            {
                posfix+=s.top();

            }
            s.push(t[i]);
